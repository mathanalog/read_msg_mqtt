#!/usr/bin/env bash

N=80

# Step in minutes for timestamp
step=60

# SO2
sensor_id=164820806
units="ppb"
for ((i=0; i<$N;i++)){
        ts=$(date -d "now + $(($i*$step)) minutes" +'%Y-%m-%d %H:%M:%S')
        value=$(( (RANDOM%100) + 40 ))
        echo "INSERT INTO measurment VALUES ($sensor_id, '$units', $value, '$ts');"
}
